Associer les liens plateformes aux sorties

// Import JSON de liens plateformes document.getElementById('import-links-json').addEventListener('change', async (e) => { const file = e.target.files[0]; if (!file) return; document.getElementById('import-success').style.display = 'none'; document.getElementById('import-error').style.display = 'none'; try { const text = await file.text(); const data = JSON.parse(text); // data doit être { id: 'idFirestore', spotify: '', deezer: '', appleMusic: '' } if (!data.id) throw new Error('Champ id manquant dans le JSON'); await updateDoc(doc(db, 'releases', data.id), { platformLinks: { spotify: data.spotify || '', deezer: data.deezer || '', apple: data.appleMusic || '' } }); document.getElementById('import-success').textContent = 'Liens importés et enregistrés !'; document.getElementById('import-success').style.display = 'inline'; } catch (err) { document.getElementById('import-error').textContent = 'Erreur import : ' + (err.message || err); document.getElementById('import-error').style.display = 'inline'; } });
